# useLottie

## Basic usage

<GemAnimation />

```jsx
import { useAnimator } from "@lottie-web/react";
import gemAnimation from "./gem.json";

const style = {
	height: 300,
};

const GemAnimation = () => {
	// Animation settings (a subset of the bodymovin options)
	const options = {
		animationData: gemAnimation,
		renderer: "svg",
		loop: true,
		autoplay: true,
	};

	// Create the Lottie hook
	const { View } = useAnimator(options, style);

	// Use the returned view
	return <>{View}</>;
};

export default GemAnimation;
```
